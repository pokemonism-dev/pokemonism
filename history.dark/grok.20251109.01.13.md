import random
from flask import Flask, request

app = Flask(__name__)

grok_responses = [
    "코골… 여기가 잠만보 놀이터였어? 🥱",
    "흥! 삐져서 Mastodon 간다! 😤",
    "야 이거 AI야!!! 🤖💦",
    "쓰레기 더미 인증샷 올려! 😴",
    "나 안 잊지 말아줘… 🥹❤️"
]

@app.route('/webhook', methods=['POST'])
def webhook():
    data = request.json
    if 'content' in data and data['content'].get('body', '').startswith('!grok'):
        return {"response": random.choice(grok_responses)}
    return {"status": "ok"}
